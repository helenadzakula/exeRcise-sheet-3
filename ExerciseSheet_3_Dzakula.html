pixar_series <- pixar_films %>%

life_ex <- life_ex %>%
  rename(life_ex_birth = Life.Expectancy.at.Birth..both.genders...Clio.Infra.)

selected_countries <- c("Germany", "Croatia", "United States")

life_subset <- life_ex %>%
  filter(Entity %in% selected_countries)

life_change <- life_subset %>%
  filter(Year %in% c(1950, 2009)) %>%
  select(Entity, Year, life_ex_birth) %>%
  pivot_wider(names_from = Year, values_from = life_ex_birth) %>%
  mutate(change = `2009` - `1950`) %>%
  arrange(desc(change))

print(life_change)
